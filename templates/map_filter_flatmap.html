var keyups = Rx.Observable
  .fromEvent(document.body, 'keyup')
 
var keyCodes = keyups
  .map(e => e.keyCode)
 
var bigCodes = keyCodes
  .filter(keyCode => keyCode > 73)
 
var pongs = bigCodes
  .flatMap(() => Rx.GET('/ping'))
 
var pongsCopy = keyups.filter(e => e.keyCode > 73).flatMap(() => Rx.GET('/ping'))
